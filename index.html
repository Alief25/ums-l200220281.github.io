Proses Kuliah dengan Metaflow

Berikut adalah kode program Python menggunakan Metaflow:

from metaflow import FlowSpec, step

class ProsesKuliahFlow(FlowSpec):

    @step
    def start(self):
        print("=== Proses Kuliah Mahasiswa ===\\n")
        print("Langkah 1: Pembayaran SPP")
        self.next(self.ambil_mata_kuliah)
    
    @step
    def ambil_mata_kuliah(self):
        print("Langkah 2: Mengambil Mata Kuliah")
        self.next(self.ikut_kuliah)

    @step
    def ikut_kuliah(self):
        print("Langkah 3: Mengikuti Perkuliahan")
        self.next(self.input_nilai)

    @step
    def input_nilai(self):
        print("Langkah 4: Menginput Nilai")
        self.next(self.nilai_akhir)

    @step
    def nilai_akhir(self):
        print("Langkah 5: Mendapatkan Nilai Akhir")
        self.next(self.end)

    @step
    def end(self):
        print("Proses kuliah telah selesai.")

if __name__ == '__main__':
    ProsesKuliahFlow()
